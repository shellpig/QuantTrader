### 型別標注

- 所有公開函式**必須**有完整的 type hint（參數 + 回傳值）
- 使用 Python 3.12 原生語法：`list[str]`、`dict[str, float]`、`str | None`

### 時區鐵律

- **所有 datetime 必須為 timezone-aware**
- 禁止出現 naive datetime
- Phase 1-8 台股資料使用 `Asia/Taipei`
- Phase 9 起改為 market-aware timezone：台股 `Asia/Taipei`，美股 `America/New_York`
- 常數定義在 `src/core/constants.py`；新增市場時不得把 timezone 寫死在 fetcher、storage、backtest 或 UI 任一層

### 設定檔讀取

- 密鑰：`.env` → 用 `python-dotenv` 讀取
- 參數：`config.yaml` → 用 `pyyaml` 讀取
- 統一透過 `src/core/config.py` 的 `get_config()` 函式存取

### 錯誤處理原則

- 外部 API 呼叫：捕獲 `requests.exceptions.RequestException`，重試最多 3 次（間隔 1/2/4 秒）
- 資料驗證失敗：拋出自定義 `DataQualityError`（定義在 `src/core/exceptions.py`）
- 內部模組間：信任輸入，不做冗餘驗證

### 測試規範

- 框架：`pytest`
- 測試檔命名：`tests/test_{module_name}.py`
- 固定測試資料放 `tests/fixtures/`（小型 CSV 或 Parquet，手工構造已知答案）
- 不 mock 資料庫（DuckDB 用 `:memory:` 模式即可）
- 外部 API 測試用 `@pytest.mark.integration` 標記，預設不跑

```python
        Args:
            symbol: 台股代碼，純數字字串，如 "2330"
            start: 起始日期 "YYYY-MM-DD"
            end: 結束日期 "YYYY-MM-DD"

        Returns:
            DataFrame，欄位為 STANDARD_COLUMNS，date 為 tz-aware Asia/Taipei
            若無資料回傳空 DataFrame（欄位仍齊全）
        """
        ...

    def fetch_minute(self, symbol: str, start: str, end: str, freq: str = "1") -> pd.DataFrame:
```

```python
@dataclass
class QualityReport:
    """品質檢核報告。"""
    symbol: str
    total_rows: int
    negative_price_count: int       # L1: 負價格
    zero_price_count: int           # L1: 零價格
    extreme_change_count: int       # L1: 漲跌超過 ±10%
    extreme_change_dates: list[str] # L1: 超限日期列表
    zero_volume_count: int          # 成交量=0 日數
    filled_count: int               # L2: 被前向填充的缺值數
    suspension_ranges: list[tuple[str, str]]  # L2: 停牌區間 [(start, end), ...]

    def summary(self) -> str:
        """回傳人類可讀的摘要文字。"""
        ...


class DataCleaner:
    """
    資料品質檢核與清洗。

    使用方式：
        cleaner = DataCleaner()
        cleaned_df, report = cleaner.clean(raw_df, symbol="2330")
    """

    def clean(self, df: pd.DataFrame, symbol: str) -> tuple[pd.DataFrame, QualityReport]:
        """
        依序執行 L1 → L2 清洗，回傳 (清洗後 DataFrame, 品質報告)。
        不修改原始 df（回傳新物件）。
        """
        ...

    def _l1_detect_anomalies(self, df: pd.DataFrame) -> tuple[pd.DataFrame, dict]:
        """
        L1 異常值偵測：
        1. 價格（open/high/low/close）為負值或 0 → 整列標記移除
        2. 單日漲跌超過 ±10% → 標記（加入 _flagged 欄位），不自動移除
           - 計算方式：(close - prev_close) / prev_close
           - 第一筆無前日資料，跳過
           - 注意：IPO 首五日豁免邏輯在 V1 先不實作，僅標記警告
        3. 成交量 = 0 → 不移除，僅記錄

        回傳：(處理後 df, 統計 dict)
        """
        ...

    def _l2_fill_missing(self, df: pd.DataFrame) -> tuple[pd.DataFrame, dict]:
        """
        L2 缺失值填補：
        1. 日K：前向填充（ffill），連續缺值 > 5 日標記停牌
        2. 識別停牌區間：連續成交量 = 0 且 ≥ 5 個交易日

        回傳：(填補後 df, 統計 dict)
```

```python
class StrategyBase(ABC):
    """
    策略基類。同時支援向量化引擎與事件驅動引擎。

    向量化引擎呼叫 generate_signals()。
    事件驅動引擎呼叫 on_bar()。
    子類別必須兩者都實作。
    """

    @abstractmethod
    def generate_signals(self, df: pd.DataFrame) -> pd.Series:
        """
        向量化訊號生成。

        Args:

        - 訊號在 Bar t 收盤後產生，意味著 signal[t] 的交易在 t+1 開盤執行
        - 不可使用未來資料（signal[t] 只能用 df[:t+1] 的資訊）

        ...

    @abstractmethod
    def on_bar(self, bar: BarEvent, account: Account) -> list[OrderEvent]:
        """事件驅動模式：每根 K 線呼叫一次。"""
        ...

    def on_fill(self, fill, account: Account) -> None:
        """訂單成交後回呼（可選覆寫）。"""
        pass
```

```python
from dataclasses import dataclass

@dataclass
class TradeCost:
    """單筆交易成本明細。"""
    commission: float     # 手續費
    tax: float            # 交易稅
    slippage: float       # 滑價成本（金額）
    total: float          # commission + tax + slippage


class CostCalculator:
    """
    台股交易成本計算器。

    參數來自 config.yaml：
    - commission_rate: 0.001425
    - commission_discount: 0.6
    - tax_rate: 0.003（一般股票）
    - etf_tax_rate: 0.001（ETF）
    - slippage_ticks: 1
    """

    def __init__(
        self,
        commission_rate: float = 0.001425,
        commission_discount: float = 0.6,
        tax_rate: float = 0.003,
        etf_tax_rate: float = 0.001,
        slippage_ticks: int = 1,
    ):
        self.effective_commission_rate = commission_rate * commission_discount
        self.tax_rate = tax_rate
        self.etf_tax_rate = etf_tax_rate
        self.slippage_ticks = slippage_ticks

    def calculate(
        self,
        price: float,
        quantity: int,
        side: str,
        is_etf: bool = False,
    ) -> TradeCost:
        """
        計算單筆交易成本。

        side: "BUY" | "SELL"
        - 手續費：price × quantity × effective_commission_rate（買賣皆收）
          手續費不足 20 元以 20 元計（台股券商規定）
        - 交易稅：僅賣出收。price × quantity × tax_rate（ETF 用 etf_tax_rate）
        - 滑價：get_tick_size(price) × slippage_ticks × quantity
        """
        ...

    @staticmethod
    def get_tick_size(price: float) -> float:
        """
        台股升降單位：
        price < 10       → 0.01
        10 ≤ price < 50  → 0.05
        50 ≤ price < 100 → 0.10
        100 ≤ price < 500 → 0.50
        500 ≤ price < 1000 → 1.00
        price ≥ 1000     → 5.00
        """
        ...

    def apply_slippage(self, price: float, side: str) -> float:
        """
        回傳滑價後的成交價。
        BUY: price + tick_size × slippage_ticks
        SELL: price - tick_size × slippage_ticks
        """
        ...
```

```python
@dataclass
class BacktestResult:
    """回測結果容器，供 report.py 使用。"""
    equity_curve: pd.Series          # 每日權益值
    returns: pd.Series               # 每日報酬率
    trades: pd.DataFrame             # 交易明細（entry_date, exit_date, side, pnl, ...）
    total_return: float
    annual_return: float
    max_drawdown: float
    max_drawdown_start: str          # 最大回撤起始日
    max_drawdown_end: str            # 最大回撤結束日
    sharpe_ratio: float
    sortino_ratio: float
    calmar_ratio: float
    win_rate: float
    profit_factor: float
    total_trades: int
    avg_holding_days: float
    max_single_loss: float


def calculate_metrics(
    equity_curve: pd.Series,
    trades: pd.DataFrame,
    risk_free_rate: float = 0.01,
    trading_days: int = 252,
) -> BacktestResult:
    """
    從權益曲線 + 交易明細計算全部績效指標。

    equity_curve: index=date, value=總資產市值（含現金+持倉市值）
    trades: 每筆完成的交易，欄位含 entry_date, exit_date, pnl

    trading_days: 台股年交易日數

    計算公式：
    - returns = equity_curve.pct_change().dropna()
    - annual_return = (equity_curve.iloc[-1] / equity_curve.iloc[0]) ^ (252/n_days) - 1
    - sharpe = (returns.mean() - rf_daily) / returns.std() × sqrt(252)
    - sortino = (returns.mean() - rf_daily) / downside_std × sqrt(252)
      downside_std = std of returns[returns < 0]（若全正則 sortino = inf → 用 999.0 代替）
    - calmar = annual_return / abs(max_drawdown)（若 MDD=0 → 999.0）
    - max_drawdown = max(1 - equity_curve / equity_curve.cummax())
    - win_rate = winning_trades / total_trades
    - profit_factor = sum(gains) / abs(sum(losses))（若無虧損 → 999.0）
    """
```

```python
class OrderEvent:
    """下單事件。"""
    symbol: str
    order_type: str              # "MARKET" | "LIMIT"
    side: str                    # "BUY" | "SELL"
    quantity: int                # 正整數（股數，台股 1 張 = 1000 股）
    price: float | None = None   # LIMIT 單必填，MARKET 單為 None

    def __post_init__(self):

            raise ValueError("LIMIT order must have a price")

            raise ValueError(f"Invalid side: {self.side}")


@dataclass(frozen=True)
class FillEvent:
    """成交事件。"""
    symbol: str
    side: str
    quantity: int
    fill_price: float
    commission: float
    tax: float
    timestamp: datetime

    @property
    def total_cost(self) -> float:
        """成交總金額（含手續費 + 稅）。"""
        base = self.fill_price * self.quantity
        if self.side == "BUY":
            return base + self.commission
        else:
            return base - self.commission - self.tax

    def to_dict(self) -> dict:
        return asdict(self)
```

```python
        """回傳指定標的的平均成本價。無持倉回傳 0.0。"""
        ...


class SimpleAccount(Account):
    """
    V1.0 單一帳戶實作。

    狀態：
    - cash: float（當前現金）
    - positions: dict[str, int]（持倉股數，正數 = 多頭）
    - cost_basis: dict[str, float]（加權平均成本價）

    apply_fill 邏輯：
    BUY:
        total = fill_price × quantity + commission
        cash -= total
        更新 positions[symbol] += quantity
        更新 cost_basis（加權平均）：
            new_cost = (old_cost × old_qty + fill_price × quantity) / (old_qty + quantity)

    SELL:
        total = fill_price × quantity - commission - tax
        cash += total
        positions[symbol] -= quantity
        若 positions[symbol] == 0 → 從 dict 移除，cost_basis 也移除
        若 positions[symbol] < 0 → 拋出 ValueError（V1 不支援放空）
    """

    def __init__(self, initial_capital: float):
        self._initial_capital = initial_capital
        self.cash = initial_capital
        self.positions: dict[str, int] = {}
        self.cost_basis: dict[str, float] = {}
        self._history: list[dict] = []    # 每筆操作紀錄，供事後分析

    def apply_fill(self, fill: FillEvent) -> None:
        ...
        # 每次操作後記錄快照到 _history
        self._history.append({
            "timestamp": fill.timestamp,
            "action": fill.side,
            "symbol": fill.symbol,
            "quantity": fill.quantity,
            "price": fill.fill_price,
            "cash_after": self.cash,
            "positions_after": dict(self.positions),
        })
```

```python
import pandas as pd
from src.backtest.base import BacktesterBase
from src.backtest.account import SimpleAccount

from src.backtest.cost import CostCalculator
from src.backtest.metrics import BacktestResult, calculate_metrics
from src.strategy.base import StrategyBase

class EventDrivenBacktester(BacktesterBase):
    """
    事件驅動回測引擎。

    核心設計：
    - 逐 Bar 迭代 data DataFrame
    - 每根 Bar：
      1. 若有上一根 Bar 的 pending_orders → 以本根 Bar 開盤價撮合
      2. 呼叫 strategy.on_bar(bar_event, account)
      3. 收到的 OrderEvent 存入 pending_orders（下根 Bar 執行）
      4. 記錄當日權益值（equity_curve）

    這確保「訊號產生於 Bar t，成交於 Bar t+1」，避免未來函數。
    """

    def __init__(
        self,
        initial_capital: float = 1_000_000,
        cost_calculator: CostCalculator | None = None,
    ):
        self.initial_capital = initial_capital
        self.cost_calculator = cost_calculator or CostCalculator()

    def run(self, strategy: StrategyBase, data: pd.DataFrame) -> BacktestResult:
        account = SimpleAccount(self.initial_capital)
        pending_orders: list[OrderEvent] = []
        equity_records: list[dict] = []
        trade_records: list[dict] = []

        bars = list(data.itertuples())

        for i, bar in enumerate(bars):
            bar_event = self._to_bar_event(bar, data)

            # Step 1: 撮合 pending_orders（以本根開盤價）
            for order in pending_orders:
                fill = self._simulate_fill(order, bar_event)
                account.apply_fill(fill)
                strategy.on_fill(fill, account)
                trade_records.append(fill.to_dict())

            pending_orders.clear()

            # Step 2: 策略產生新訂單
            new_orders = strategy.on_bar(bar_event, account)
            if new_orders:
                pending_orders.extend(new_orders)

            # Step 3: 記錄權益
            current_prices = {bar_event.symbol: bar_event.close}
            equity_records.append({
                "date": bar_event.timestamp,
                "equity": account.get_total_value(current_prices),
            })

        return self._build_result(equity_records, trade_records)

    def _to_bar_event(self, bar, data: pd.DataFrame) -> BarEvent:
        """將 DataFrame row 轉為 BarEvent。"""
        ...

    def _simulate_fill(self, order: OrderEvent, bar: BarEvent) -> FillEvent:
        """
        撮合模擬：
        - MARKET 單：以 bar.open ± slippage 成交
        - LIMIT 單：若 bar.low ≤ price ≤ bar.high 則成交，否則不成交
          （V1 先只實作 MARKET 單）
        - 計算 commission + tax
        - 回傳 FillEvent
        """
        ...

    def _build_result(self, equity_records, trade_records) -> BacktestResult:
        ...
```

```python
from dataclasses import dataclass
from typing import Literal

Market = Literal["tw", "us"]

@dataclass(frozen=True)
```

### B. AIAdvisor 改為 provider-neutral

`AIAdvisor` 應負責投資分析流程與工具調度，不應直接綁定 Claude SDK。建議拆成：

```text
src/ai/
  advisor.py
  providers/
    base.py
    openai_provider.py
    anthropic_provider.py
    gemini_provider.py
  tools.py
```

核心介面建議：

```python
class LLMProvider(Protocol):
    def ask(
        self,
        messages: list[dict],
        tools: list[dict],
        model: str,
        temperature: float,
        max_tokens: int,
    ) -> LLMResponse:
```

#### `src/strategy/examples/rsi.py`

```python
class RSIStrategy(StrategyBase):
    def __init__(
        self,
        period: int = 14,
        oversold: float = 30.0,
        overbought: float = 70.0,
    ) -> None: ...

    def generate_signals(self, df: pd.DataFrame) -> pd.Series: ...
    def on_bar(self, bar: BarEvent, account: Account) -> list[OrderEvent]: ...
    def reset_runtime_state(self) -> None: ...
```

**向量化邏輯：**
1. 以 `pandas_ta.rsi(close, length=period)` 計算 RSI。
2. 追蹤 RSI 是否低於 `oversold`，回升到 `oversold` 以上時發 +1。
3. 追蹤 RSI 是否高於 `overbought`，回落到 `overbought` 以下時發 -1。
4. Warm-up 期（前 `period` 筆 RSI 為 NaN）訊號為 0。

**事件驅動邏輯：**
1. 維護 `_close_history: list[float]`，逐 bar 累計。
2. 以 Wilder smoothing 手動計算 RSI（或累積足夠 bar 後用 pandas_ta）。
3. 比較前一 bar 的 RSI 與當前 RSI 判斷上穿/下穿。
4. 買進用 `_BUY_INTENT_QUANTITY`，賣出用 `account.get_position(bar.symbol)`。

**參數驗證：**
- `period > 0`
- `0 <= oversold < overbought <= 100`

#### `src/strategy/examples/kd_cross.py`

```python
class KDCrossStrategy(StrategyBase):
    def __init__(
        self,
        k_period: int = 9,
        d_period: int = 3,
        smooth_k: int = 3,
    ) -> None: ...
```

**向量化邏輯：**
1. 以 `pandas_ta.stoch(high, low, close, k=k_period, d=d_period, smooth_k=smooth_k)` 計算 K、D。
2. K 從下方上穿 D → +1（黃金交叉）。
3. K 從上方下穿 D → -1（死亡交叉）。
4. 判斷交叉：`(K_prev <= D_prev) & (K_curr > D_curr)` 為上穿。

**事件驅動邏輯：**
1. 維護 `_high_history`、`_low_history`、`_close_history`。
2. 累積足夠 bar 後計算 Stochastic K 和 D。
3. 比較前一 bar 與當前的 K-D 關係判斷交叉。

**Warm-up：** 前 `k_period + smooth_k + d_period - 2` 筆訊號為 0。

**參數驗證：**
- `k_period > 0, d_period > 0, smooth_k > 0`

```python
def _build_strategy(preset: dict) -> StrategyBase:
    """根據 preset 的 type 和 params 建構策略 instance。

    抽出來讓 batch.py、sweep.py、backtest.py UI 三處共用，
    避免策略建構邏輯散布三處。
    """
    strategy_type = preset.get("type", "")
    params = preset.get("params", {})

    builders: dict[str, Callable] = {
        "moving_average_cross": lambda p: MACrossStrategy(**p),
        "rsi": lambda p: RSIStrategy(**p),
        "kd_cross": lambda p: KDCrossStrategy(**p),
        "macd_cross": lambda p: MACDCrossStrategy(**p),
        "bollinger_band": lambda p: BollingerBandStrategy(**p),
        "bias": lambda p: BiasStrategy(**p),
        "donchian_breakout": lambda p: DonchianBreakoutStrategy(**p),
    }

    builder = builders.get(strategy_type)
    if builder is None:
        raise ValueError(f"Unsupported strategy type: {strategy_type}")
    return builder(params)
```

```python
def generate_param_grid(
    strategy_type: str,
    param_ranges: dict[str, list],
) -> tuple[list[dict[str, Any]], int]:
    """生成參數組合，過濾不合法的。

    回傳 (合法組合 list, 過濾前總組合數)。
    """
    from src.core.strategy_config import (
        _normalize_rsi_params,
        _normalize_kd_cross_params,
        _normalize_macd_cross_params,
        _normalize_bollinger_band_params,
        _normalize_bias_params,
        _normalize_donchian_breakout_params,
        _normalize_moving_average_params,
    )

    normalizers = {
        "moving_average_cross": _normalize_moving_average_params,
        "rsi": _normalize_rsi_params,
        "kd_cross": _normalize_kd_cross_params,
        "macd_cross": _normalize_macd_cross_params,
        "bollinger_band": _normalize_bollinger_band_params,
        "bias": _normalize_bias_params,
        "donchian_breakout": _normalize_donchian_breakout_params,
    }

    normalizer = normalizers.get(strategy_type)
    if normalizer is None:
        raise ValueError(f"Unsupported strategy type for sweep: {strategy_type}")

    keys = list(param_ranges.keys())
    values = list(param_ranges.values())
    all_combos = [dict(zip(keys, combo)) for combo in product(*values)]
    total = len(all_combos)

    valid = []
    for combo in all_combos:
        if normalizer(combo) is not None:
            valid.append(combo)

    return valid, total
```

```python
def run_parameter_sweep(...) -> SweepResult:
    valid_combos, total_combos = generate_param_grid(strategy_type, param_ranges)

    if len(valid_combos) > MAX_SWEEP_COMBOS:
        raise ValueError(
            f"合法組合數 {len(valid_combos)} 超過上限 {MAX_SWEEP_COMBOS}，請縮小參數範圍。"
        )

    results = []
    for combo in valid_combos:
        preset = {"type": strategy_type, "params": combo, "name": "sweep"}
        try:
            strategy = _build_strategy(preset)
            engine = VectorizedBacktester(initial_capital=initial_capital)
            bt_result = engine.run(strategy=strategy, data=data)

            summary = SweepRunSummary(
                params=combo,
                total_return=bt_result.total_return,
                annual_return=bt_result.annual_return,
                max_drawdown=bt_result.max_drawdown,
                sharpe_ratio=bt_result.sharpe_ratio,
                win_rate=bt_result.win_rate,
                profit_factor=bt_result.profit_factor,
                total_trades=bt_result.total_trades,
                sample_warning=(bt_result.total_trades < 3),
                result=bt_result,
            )
        except Exception as e:
            summary = SweepRunSummary(
                params=combo,
                total_return=0, annual_return=0, max_drawdown=0,
                sharpe_ratio=0, win_rate=0, profit_factor=0,
                total_trades=0, error=str(e),
            )
        results.append(summary)

    return SweepResult(
        symbol=symbol,
        start_date=str(data.index[0].date()),
        end_date=str(data.index[-1].date()),
        strategy_type=strategy_type,
        total_combos=total_combos,
        valid_combos=len(valid_combos),
        results=results,
    )
```

```python
def _parse_param_values(raw: str, param_key: str) -> list:
    """解析逗號分隔的參數值。

    數值型參數轉為 int 或 float；去重並排序。
    """
    parts = [p.strip() for p in raw.split(",") if p.strip()]
    values = []
    for p in parts:
        try:
            values.append(int(p))
        except ValueError:
            values.append(float(p))
    return sorted(set(values))
```

```python
def _select_best_sweep_run(
    sweep_runs: list[SweepRunSummary],
    optimize_metric: str,
) -> SweepRunSummary | None:
    candidates = [
        r for r in sweep_runs
        if r.error is None and r.result is not None
    ]
    if not candidates:
        return None
    return max(candidates, key=lambda r: getattr(r, optimize_metric))
```

```python
def _calculate_short_term_score(
    *,
    ma_score: float,      # 0.0~1.0
    kd_score: float,      # 0.0~1.0
    volume_price_score: float,  # 0.0~1.0
    breakout_score: float,      # 0.0~1.0
) -> tuple[float, str, dict[str, float]]:
    """加權計算短線綜合分數。

    權重：MA 結構 30%, KD 狀態 25%, 量價關係 25%, 突破狀態 20%。
    """
    score = (
        ma_score * 0.30
        + kd_score * 0.25
        + volume_price_score * 0.25
        + breakout_score * 0.20
    )
    if score >= 0.7:
        label = "強勢偏多"
    elif score >= 0.5:
        label = "中等偏多"
    elif score >= 0.3:
        label = "中性"
    else:
        label = "偏空"
    components = {
        "ma": ma_score,
        "kd": kd_score,
        "volume_price": volume_price_score,
        "breakout": breakout_score,
    }
    return score, label, components
```

```python
            ]
```

```python
            st.success(f"{pattern.pattern_type}：{pattern.description}")
```

## Phase 9：美股 US-1 支援

**對應規格：** 規格書 §11 Phase 9。

**總目標：** 在不破壞台股既有流程的前提下，新增美股第一版支援。9-A 到 9-F 的 US-1 只做美股日 K、調整後價格、回測與技術分析；9-G 擴充 yfinance 1m intraday 盤中快照與分 K 圖。Phase 9 仍不做籌碼、財報、匯率換算、期權、WebSocket、買一 / 賣一、五檔、逐筆或實盤級即時報價。

**實作總原則：**

- `market` 是所有市場差異的第一層分流，預設值為 `tw`。
- 台股既有功能不得因美股擴充改變預設行為。
- 美股所有日期以 `America/New_York` timezone 儲存，且必須 timezone-aware。
- 美股回測與報表以 USD 呈現，不換算 TWD。
- 美股成交量顯示 shares，不轉台股「張」。
- 美股資料源第一版只用 `yfinance`，定位為個人研究用途。
- 9-G 盤中價使用 yfinance 最新 1 分 K raw close，漲跌基準使用前一紐約交易日 raw close；必須標示可能延遲且僅供研究分析。
- 美股 UI 放在既有回測頁、個股分析頁與資料管理頁，以市場切換控制，不新增獨立美股頁。
- 單一執行流程只能有一個 `market` context；US-1 禁止混合台股與美股 DataFrame 做 concat、join、align 或同一份 payload 分析。

```python
def assert_single_market(markets: Iterable[str]) -> str:
    normalized = {normalize_market(m) for m in markets}
    if len(normalized) != 1:
        raise ValueError(f"Mixed market context is not supported in US-1: {sorted(normalized)}")
    return next(iter(normalized))
```

使用位置：

- 資料管理頁：單次更新 / 重建必須同一 market。

#### 美股 adjusted daily 產生規則

yfinance raw 欄位若包含 `Adj Close`：

```python
price_ratio = raw["Adj Close"] / raw["Close"]
adjusted["open"] = raw["Open"] * price_ratio
adjusted["high"] = raw["High"] * price_ratio
adjusted["low"] = raw["Low"] * price_ratio
adjusted["close"] = raw["Adj Close"]
```

成交量必須做 split-adjust，不可保留 raw volume 直接進 adjusted daily：

```python
adjusted["volume"] = raw["Volume"] * cumulative_split_factor
```

重點：

- `price_ratio = Adj Close / Close` 可能包含 dividend adjustment，只能用來調 OHLC。
- `cumulative_split_factor` 必須只來自 split events，不能混入 dividend adjustment。
- 若 yfinance downloader 支援 `actions=True` 並回傳 `Stock Splits`，優先用該欄位建立 cumulative split factor。
- 若實作改走 `Ticker.splits`，需確保 split 日期與 daily index 以 `America/New_York` 對齊。
- 對 4-for-1 split，split 前的歷史 volume 應乘 4，與 split-adjusted price 對齊。

- 若 `Close <= 0` 或 ratio 無法計算，該列應丟棄或 fallback，不能產生 inf / NaN 價格。
- 若整批資料沒有 `Adj Close`，可用 raw OHLCV 存 adjusted，但 caller 需要拿到狀態或 log warning：「使用未調整價格」。
- 若無法取得 split events 或無法確認 split factor，不能宣稱 volume 已完成 split-adjust。Dashboard 的 OBV、量能放大 / 縮小、量價結構需顯示資料限制提示，或改用 raw volume 並明確標示限制。

#### yfinance 節流

單檔更新不需額外 sleep。批次更新 `market="us"` 時，maintenance 層必須在每個 symbol request 間 sleep 至少 1 秒：

```python
US_YFINANCE_REQUEST_INTERVAL_SECONDS = 1.0
```

US-1 可先用常數，不必加入 `config.yaml`。遇到 429 / timeout 時需回報外部資料源限制，不得把 symbol 標成永久無資料。

#### 時區

## Phase 10：前端架構重構 — Streamlit → Next.js

**對應規格：** 規格書 Phase 10。

**總目標：** 從 Streamlit 遷移至 Next.js + FastAPI，解決佈局、手機適配、效能與互動性限制。核心演算法 (`src/analysis/`, `src/backtest/`, `src/data/`, `src/strategy/`, `src/core/`, `src/ai/`, `src/indicators/`) 不重寫；新增 `src/services/` 服務層、`api/` 後端、`web/` 前端。

- JSON 序列化使用 `dataclasses.asdict()` 或 Pydantic model。
- 統一回應 envelope：`{ "data": {...}, "meta": {...} }` 或 `{ "error": { "code": "...", "message": "..." } }`。
- Envelope 例外：`GET /api/health`、`POST /api/jobs`、`GET /api/jobs/{id}/events` (SSE)。
- CORS 開發環境允許 `localhost:3000`。
- Secrets 安全：GET 永不回傳 API key 值；寫入走 write-only endpoint；config 更新走 schema whitelist。
- 前端使用 client-side fetching (SWR)，不做 Server Components streaming。

```typescript
// web/src/lib/api-client.ts

const BASE_URL = process.env.NEXT_PUBLIC_API_URL || "http://localhost:8000";

interface ApiResponse<T> {
  data: T;
  meta: { market: string; symbol?: string; timestamp: string };
}

interface ApiError {
  error: { code: string; message: string };
}

export async function apiFetch<T>(path: string, options?: RequestInit): Promise<T> {
  const res = await fetch(`${BASE_URL}${path}`, {
    ...options,
    headers: { "Content-Type": "application/json", ...options?.headers },
  });
  if (!res.ok) {
    const body = await res.json().catch(() => null);
    throw new ApiClientError(res.status, body?.error?.code, body?.error?.message);
  }
  return res.json();
}
```

#### Lightweight Charts React wrapper 模式

```tsx
// web/src/components/charts/candlestick-chart.tsx

import { createChart, ColorType, IChartApi } from "lightweight-charts";
import { useEffect, useRef } from "react";

interface CandlestickChartProps {
  data: { time: string; open: number; high: number; low: number; close: number }[];
  market: "tw" | "us";
}

export function CandlestickChart({ data, market }: CandlestickChartProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const chartRef = useRef<IChartApi | null>(null);

  useEffect(() => {
    if (!containerRef.current) return;

    const upColor = market === "tw" ? "#ef4444" : "#22c55e";
    const downColor = market === "tw" ? "#22c55e" : "#ef4444";

    const chart = createChart(containerRef.current, {
      width: containerRef.current.clientWidth,
      height: 400,
      layout: { background: { type: ColorType.Solid, color: "transparent" } },
      crosshair: { mode: 0 },
      timeScale: { timeVisible: false },
    });

    const series = chart.addCandlestickSeries({
      upColor, downColor, wickUpColor: upColor, wickDownColor: downColor,
    });
    series.setData(data);
    chart.timeScale().fitContent();
    chartRef.current = chart;

    const handleResize = () => {
      chart.applyOptions({ width: containerRef.current!.clientWidth });
    };
    window.addEventListener("resize", handleResize);

    return () => {
      window.removeEventListener("resize", handleResize);
      chart.remove();
    };
  }, [data, market]);

  return <div ref={containerRef} className="w-full" />;
}
```
